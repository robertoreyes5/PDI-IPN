from google.colab.patches import cv2_imshow
import cv2
import numpy as np

imge1 = cv2.imread("/content/imagen1.png")
imge2 = cv2.imread("/content/imagen2.png")


#img1 = cv2_imshow(imge1)
#img2 = cv2_imshow(imge2)
gray1 = cv2.cvtColor(imge1, cv2.COLOR_BGR2GRAY)
gray2 = cv2.cvtColor(imge2, cv2.COLOR_BGR2GRAY)

diff = cv2.absdiff(gray1, gray2)
_, thresh = cv2.threshold(diff, 25, 255, cv2.THRESH_BINARY)

contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

max_area = 0
best_contour = None
for contour in contours:
    area = cv2.contourArea(contour)
    if area > max_area:
        max_area = area
        best_contour = contour

cv2.drawContours(imge2, [best_contour], -1, (0, 255, 0), 2)
cv2_imshow( imge2)
cv2.waitKey(0)
cv2.destroyAllWindows()
